```javascript
  createdAt: Date,
```

```javascript
{
  _id: ObjectId,

  updatedAt: Date
}
```

```javascript
{
  _id: ObjectId,

  bookingNumber: String (unique),
  date: Date,
  startTime: Date,
  endTime: Date,

    firstName: String,

    amount: Number,

  },

    emailSent: Boolean,
    smsSent: Boolean,

    smsSentAt: Date
  },

    reason: String,

    cancelledAt: Date,
    refundStatus: String
  },
  createdAt: Date,
  updatedAt: Date
```

```javascript
{
  _id: ObjectId,
```

```javascript
{
```
